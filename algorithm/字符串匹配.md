# 字符串匹配

kmp

```c
#include <stdio.h>
#include <memory.h>
#include <stdlib.h>

void getNext(char* s,int* n){
    n[0]=0;
    for(int i=1,k=0;i<strlen(s);i++){
        while(k>0&&s[k]!=s[i])
            k=n[k-1];
        if(s[k]==s[i]) k++;
        n[i]=k;
    }
}

int kmp(char* s1,char *s2){
    int len2 = strlen(s2);
    int* n=(int*)calloc(strlen(s2),sizeof(int));
    getNext(s2,n);
    for(int i=0,j=0;i<strlen(s1);i++){
        while (j>0&&s1[i]!=s2[j])
            j=n[j-1];
        if(s1[i]==s2[j])
            j++;
        if(j==len2) {
            free(n);
            return i - j + 1;
        }
    }
    free(n);
    return -1;
}

int main() {
    char s1[]="abdbcasda";
    char s2[]="abca";
    int n = kmp(s1,s2);
    printf("%d",n);
    return 0;
}
```
